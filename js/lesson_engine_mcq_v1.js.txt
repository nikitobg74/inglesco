// lesson_engine_mcq_v1.js
// One engine for all MCQ lessons (A then B), mobile-first.
// Language-neutral: no Spanish/English UI text inside.
// Requires specific HTML ids (see below).

(function () {
  "use strict";

  function $(id) { return document.getElementById(id); }

  function safePlay(audio) {
    try {
      audio.currentTime = 0;
      const p = audio.play();
      if (p && typeof p.catch === "function") p.catch(() => {});
    } catch (_) {}
  }

  function setStatus(el, state) {
    el.classList.remove("ok", "bad");
    el.textContent = "";
    if (state === "ok") { el.classList.add("ok"); el.textContent = "✓"; }
    if (state === "bad") { el.classList.add("bad"); el.textContent = "✖"; }
  }

  function markWrong(btn) {
    btn.classList.add("wrong");
    setTimeout(() => btn.classList.remove("wrong"), 550);
  }

  function showCorrect(container, correctValue) {
    [...container.querySelectorAll("button")].forEach(btn => {
      if (btn.textContent === correctValue) btn.classList.add("correct");
      btn.disabled = true;
    });
  }

  function renderOptions(container, opts, onPick) {
    container.innerHTML = "";
    opts.forEach((text) => {
      const b = document.createElement("button");
      b.type = "button";
      b.className = "opt";
      b.textContent = text;
      b.addEventListener("click", () => onPick(b, text));
      container.appendChild(b);
    });
  }

  function setAllDisabled(container, disabled) {
    [...container.querySelectorAll("button")].forEach(btn => { btn.disabled = disabled; });
  }

  // Public API
  window.InglescoMCQ = function initEngine(config) {
    // Required config:
    // {
    //   IMG_BASE, AUDIO_BASE, waitAfterCorrectMs,
    //   manualFirstRound: true/false,
    //   rounds: [{img,audio,lines:[...], A:{prompt,options,correct}, B:{...}}]
    // }

    const CFG = Object.assign({
      waitAfterCorrectMs: 2000,
      manualFirstRound: true
    }, config || {});

    const els = {
      sceneImg: $("sceneImg"),
      playBtn: $("playBtn"),
      roundLabel: $("roundLabel"),
      line1: $("line1"),
      line2: $("line2"),
      line3: $("line3"),

      qABox: $("qABox"),
      qBBox: $("qBBox"),
      qAText: $("qAText"),
      qBText: $("qBText"),
      qAOptions: $("qAOptions"),
      qBOptions: $("qBOptions"),
      qAStatus: $("qAStatus"),
      qBStatus: $("qBStatus"),

      feedbackOk: $("feedbackOk"),
      endScreen: $("endScreen")
    };

    const audio = new Audio();
    let idx = 0;
    let locked = false;
    let allowAutoplay = !CFG.manualFirstRound; // if manualFirstRound true -> starts false

    function clearFeedback() {
      els.feedbackOk.classList.add("hidden");
      setStatus(els.qAStatus, null);
      setStatus(els.qBStatus, null);
      els.qABox.style.borderColor = "transparent";
      els.qBBox.style.borderColor = "transparent";
    }

    function flashCorrect() {
      els.feedbackOk.classList.remove("hidden");
      setTimeout(() => els.feedbackOk.classList.add("hidden"), 900);
    }

    function loadRound(i) {
      idx = i;
      locked = false;

      els.endScreen.classList.add("hidden");
      clearFeedback();

      const r = CFG.rounds[idx];
      els.roundLabel.textContent = `${idx + 1} / ${CFG.rounds.length}`;

      els.sceneImg.src = CFG.IMG_BASE + r.img;
      audio.src = CFG.AUDIO_BASE + r.audio;

      els.line1.textContent = r.lines?.[0] || "";
      els.line2.textContent = r.lines?.[1] || "";
      els.line3.textContent = r.lines?.[2] || "";

      // A
      els.qAText.textContent = `A. ${r.A.prompt}`;
      renderOptions(els.qAOptions, r.A.options, (btn, chosen) => pickA(btn, chosen));

      // B reset/hidden
      els.qBBox.classList.add("hidden");
      els.qBOptions.innerHTML = "";
      els.qBText.textContent = "B.";

      // Autoplay rule
      if (idx === 0 && CFG.manualFirstRound && !allowAutoplay) {
        // do nothing
      } else {
        safePlay(audio);
      }
    }

    function pickA(btn, chosen) {
      if (locked) return;
      const r = CFG.rounds[idx];

      if (chosen !== r.A.correct) {
        setStatus(els.qAStatus, "bad");
        markWrong(btn);
        return;
      }

      locked = true;
      setStatus(els.qAStatus, "ok");
      els.qABox.style.borderColor = "rgba(22,163,74,.55)";
      showCorrect(els.qAOptions, r.A.correct);
      flashCorrect();

      // show B
      els.qBBox.classList.remove("hidden");
      els.qBText.textContent = `B. ${r.B.prompt}`;
      renderOptions(els.qBOptions, r.B.options, (b, chosenB) => pickB(b, chosenB));

      locked = false;
    }

    function pickB(btn, chosen) {
      if (locked) return;
      const r = CFG.rounds[idx];

      if (chosen !== r.B.correct) {
        setStatus(els.qBStatus, "bad");
        markWrong(btn);
        return;
      }

      locked = true;
      setStatus(els.qBStatus, "ok");
      els.qBBox.style.borderColor = "rgba(22,163,74,.55)";
      showCorrect(els.qBOptions, r.B.correct);
      flashCorrect();

      setTimeout(() => {
        if (idx + 1 < CFG.rounds.length) loadRound(idx + 1);
        else finish();
      }, CFG.waitAfterCorrectMs);
    }

    function finish() {
      setAllDisabled(els.qAOptions, true);
      setAllDisabled(els.qBOptions, true);
      els.endScreen.classList.remove("hidden");
    }

    els.playBtn?.addEventListener("click", () => {
      allowAutoplay = true;        // after user clicks once, autoplay allowed going forward
      safePlay(audio);
    });

    loadRound(0);

    return { reload: loadRound };
  };
})();
